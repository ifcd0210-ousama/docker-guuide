Your First Dockerfile (2026 Best Practices)

Dockerfile dockerfile
Dockerfile
.dockerignore
123456789101112131415161718192021222324252627 
# Use a specific version tag — never 'latest' in production
FROM node:22-alpine AS base

WORKDIR /app

# Copy dependency manifests first for better layer caching
COPY package*.json ./

# --omit=dev replaces deprecated --only=production (npm v9+)
RUN npm ci --omit=dev && \
    npm cache clean --force

COPY . .

# Create a non-root user (security best practice)
RUN addgroup -g 1001 -S nodejs && \
    adduser -S nodejs -u 1001

USER nodejs

EXPOSE 3000

# Use wget (available in alpine) instead of node healthcheck.js
HEALTHCHECK --interval=30s --timeout=5s --start-period=10s --retries=3 \
  CMD wget -qO- http://localhost:3000/health || exit 1

CMD ["node", "server.js"]

docker rm myapp
docker rmi myapp:1.0

# ── Cleanup (removes stopped containers, dangling images) ──
# Warning: -a removes ALL unused images, not just dangling ones

docker system prune -a --volumes       # also removes all unused images

Week 6

Security with Docker Scout

scout-scan.sh bash
scout-scan.sh
123456789101112131415161718 
# Quick summary: CVE count, base image score, recommendations
docker scout quickview myapp:1.0

# Full CVE list (reachability-aware — ignores unused code paths)
docker scout cves myapp:1.0

# Filter to critical/high only
docker scout cves --only-severity critical,high myapp:1.0

# AI-prioritized remediation: base image upgrades, package fixes
docker scout recommendations myapp:1.0

# Compare two image versions (useful after patching)
docker scout compare myapp:1.0 --to myapp:1.1

# Generate an SBOM (Software Bill of Materials)
docker scout sbom myapp:1.0 --format spdx

# Enroll image into Scout org-level tracking (optional)
docker scout enroll myorg
Bash UTF-8 LF
Pro Tip 2026
Docker Scout prioritizes CVEs based on whether the vulnerable code is actually reachable in your image, dramatically reducing alert noise compared to naive scanners.

CI/CD with GitHub Actions

.github/workflows/build.yml yaml
build.yml
12345678910111213141516171819202122232425262728293031323334353637383940 
name: Docker Build & Deploy

on:
  push:
    branches: [main]

# Required for GITHUB_TOKEN to push to GHCR
permissions:
  contents: read
  packages: write
  security-events: write

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
    - uses: actions/checkout@v4

    - name: Set up Docker Buildx
      uses: docker/setup-buildx-action@v3

        # Required: authenticate before pushing to GHCR
    - name: Log in to GHCR
      uses: docker/login-action@v3
      with:
        registry: ghcr.io
        username: ${{ github.actor }}
        password: ${{ secrets.GITHUB_TOKEN }}

    - name: Build and push
      uses: docker/build-push-action@v6
      with:
        context: .
        push: true
        cache-from: type=gha
        cache-to: type=gha,mode=max
        tags: ghcr.io/${{ github.repository }}:${{ github.sha }}

    - name: Docker Scout CVE Scan
      uses: docker/scout-action@v1
      with:
        command: cves
        only-severities: critical,high
        image: ghcr.io/${{ github.repository }}:${{ github.sha }}

Image Optimization (2026 Techniques)

Dockerfile dockerfile
Dockerfile
123456789101112131415161718192021222324 
# ── Stage 1: Build ─────────────────────────────────────────
FROM node:22-alpine AS builder
WORKDIR /app
COPY package*.json ./
RUN npm ci
COPY . .
RUN npm run build
# --omit=dev replaces deprecated --production flag (npm v9+)
RUN npm prune --omit=dev

# ── Stage 2: Production ────────────────────────────────────
FROM node:22-alpine AS production

# dumb-init: lightweight PID 1 — handles signals & zombie reaping
RUN apk add --no-cache dumb-init

WORKDIR /app
# Copy only build artefacts from builder stage
COPY --from=builder /app/dist ./dist
COPY --from=builder /app/node_modules ./node_modules

# 'node' user ships in node:22-alpine — no need to create one
USER node
EXPOSE 3000
HEALTHCHECK --interval=30s --timeout=5s --retries=3 \
  CMD wget -qO- http://localhost:3000/health || exit 1
CMD ["dumb-init", "node", "dist/server.js"]
Dockerfile UTF-8 LF

services:
  prometheus:
    image: prom/prometheus:v3.2.0
    volumes:
      - ./prometheus.yml:/etc/prometheus/prometheus.yml:ro
      - prometheus-data:/prometheus
    ports:
      - "9090:9090"
    command:
      - '--config.file=/etc/prometheus/prometheus.yml'
      - '--storage.tsdb.retention.time=15d'
    restart: unless-stopped

  grafana:
    image: grafana/grafana:11.5.2
    ports:
      - "3000:3000"
    environment:
      # Override the default admin/admin credentials
      - GF_SECURITY_ADMIN_USER=admin
      - GF_SECURITY_ADMIN_PASSWORD=${GRAFANA_PASSWORD:-changeme}
      - GF_USERS_ALLOW_SIGN_UP=false
    volumes:
      - grafana-data:/var/lib/grafana
    depends_on:
      - prometheus
    restart: unless-stopped

  cadvisor:
    image: gcr.io/cadvisor/cadvisor:v0.49.2
    ports:
      - "8080:8080"
        # privileged + volumes required for full container metrics
    privileged: true
    volumes:
      - /var/run:/var/run:ro
      - /sys:/sys:ro
      - /var/lib/docker/:/var/lib/docker:ro
    restart: unless-stopped

volumes:
  prometheus-data:
